import torch
import torch.nn as nn 
from torch.utils.data import DataLoader, Dataset 
# -------------------------- 1. 定义超参数（核心概念在这里！） -------------------------- 
total_samples = 10000  # 整个数据集的总样本数（比如1万张图片） 
batch_size = 32        # 【 核心：batch_size】一次喂给模型的样本数量 
epochs = 5             # 【 核心：epoch】整个数据集完整过一遍的次数 
# -------------------------- 2. 准备数据和模型 -------------------------- 
# 模拟一个简单的数据集 
class FakeDataset(Dataset): 
    def __len__(self): 
        return total_samples 
    def __getitem__(self, idx): 
        return torch.randn(10), torch.randint(0, 2, (1,))  # 模拟特征和标签
dataset = FakeDataset() 
# 【核心：DataLoader自动帮你分batch】
# shuffle=True：每个epoch开始前打乱数据，防止模型记住顺序 
dataloader = DataLoader(dataset, batch_size=batch_size, shuffle=True) 
model = nn.Linear(10, 2)  # 简单的线性模型
criterion = nn.CrossEntropyLoss() 
optimizer = torch.optim.Adam(model.parameters(), lr=0.001)

# -------------------------- 3. 训练循环（所有概念在这里交汇！） -------------------------- 
print(f"总样本数: {total_samples}") 
print(f"单批大小 batch_size: {batch_size}") 
print(f"每个epoch的batch数（迭代次数）: { len(dataloader)}") 
print(f"总训练轮数 epochs: {epochs}") 
print(f"总迭代次数 total_iterations: {epochs * len(dataloader)}") 
print("-" * 50)

for epoch in range(epochs): 
    print(f"\n=== 开始第 {epoch+1} 个 Epoch ===")
    
    # 【核心：这个循环就是遍历所有 batch】
    # 每一次循环，dataloader会吐出1个batch的数据

        # 此时：

        # 1. 前向传播
        outputs = model(batch_data)

        loss.backward()
        optimizer.step()
        
        # 3. 迭代次数+1
        global_step += 1 
        
        # 每100个batch打印一次日志
        if batch_idx % 100 == 0: 
            print(f"Epoch: {epoch+1}/{epochs} | " 
                  f"Batch: {batch_idx}/{len(dataloader)} | " 
                  f"Global Step: {global_step} | "

单卡常见误区
1.
  错误：batch_size越大越好
  正确：受限于GPU显存，越大梯度越稳定，但训练速度越慢；常用值：32/64/128（2的幂， GPU-
计算效率更高）
2.
  错误：epoch越多越好
  正确：过多会过拟合，一般通过验证集损失判断何时停止
3.
  错误：global_step = epoch
  正确：global_step = epoch × 每个epoch的batch数

import torch
import torch.nn as nn 
import torch.distributed as dist 
import torch.multiprocessing as mp 
from torch.utils.data import DataLoader, Dataset, DistributedSampler 
# -------------------------- 1. 定义超参数 -------------------------- 
total_samples = 10000 
batch_size = 32  # 【 重要！】这里是单卡的 batch_size，不是全局的！
epochs = 5
# -------------------------- 2. 分布式训练核心函数 -------------------------- 
# 这个函数会在每个 GPU上独立运行一次
def train(rank, world_size): 
    # -------------------------- 【 Rank 核心概念初始化】 -------------------------- 
    # rank = 当前进程的全局唯一编号（ global_rank）
    # world_size = 总进程数 = 总GPU数量
    print(f"启动进程: global_rank = {rank}, world_size = {world_size}") 
    
    # 初始化分布式环境 
    dist.init_process_group(backend="nccl" , init_method="tcp://127.0.0.1:23456" , 
                            rank=rank, world_size=world_size) 
    
    # local_rank = 本机内的GPU编号（如果是单机多卡， local_rank = global_rank） 
    # 作用：指定当前进程使用哪块 GPU
    local_rank = rank % torch.cuda.device_count() 
    torch.cuda.set_device(local_rank) 
    device = torch.device(f"cuda:{local_rank}") 
    
    # -------------------------- 3. 准备数据（分布式专属！） -------------------------- 
    dataset = FakeDataset() 
    # 【核心：DistributedSampler】 自动把数据集分成 world_size份
    # 每个rank只会拿到自己那一份数据，互不重叠 
    sampler = DistributedSampler(dataset, shuffle=True) 
    
    # 注意：分布式训练时， shuffle=False，由sampler负责打乱
    dataloader = DataLoader(dataset, batch_size=batch_size, sampler=sampler) 
    
    # -------------------------- 4. 模型包装为DDP -------------------------- 
    model = nn.Linear(10, 2).to(device) 
    model = nn.parallel.DistributedDataParallel(model, device_ids=[local_rank]) 
    
    criterion = nn.CrossEntropyLoss()

    optimizer = torch.optim.Adam(model.parameters(), lr=0.001) 
    
    # -------------------------- 5. 分布式训练循环 -------------------------- 
    global_step = 0
    
    for epoch in range(epochs): 
        # 【重要！】每个epoch开始前必须调用，保证数据打乱正确 
        sampler.set_epoch(epoch) 
        
        # 只让 rank 0 打印主日志，避免多个进程重复打印 
        if rank == 0:
            print(f"\n=== 开始第 {epoch+1} 个 Epoch ===")
        
        for batch_idx, (batch_data, batch_labels) in enumerate(dataloader): 
            batch_data = batch_data.to(device) 
            batch_labels = batch_labels.to(device) 
            
            # 每个rank处理自己的batch数据
            # rank 0 处理第1份数据， rank 1 处理第2份，以此类推 
            outputs = model(batch_data) 
            loss = criterion(outputs, batch_labels.squeeze()) 
            
            optimizer.zero_grad() 
            loss.backward()
            # 【DDP核心】自动在所有 rank之间同步梯度，取平均值 
            optimizer.step()
            
            global_step += 1 
            
            # 只让 rank 0 打印日志和保存模型 
            if rank == 0 and batch_idx % 100 == 0: 
                print(f"Epoch: {epoch+1}/{epochs} | " 
                      f"Batch: {batch_idx}/{len(dataloader)} | " 
                      f"Global Step: {global_step} | " 
                      f"Loss: {loss.item():.4f}") 
    
    # 清理分布式环境
    dist.destroy_process_group() 
# -------------------------- 6. 启动分布式训练 -------------------------- 
if __name__ == "__main__": 
    world_size = torch.cuda.device_count()  # 假设用4块GPU，world_size=4

分布式训练 Batch 概念的关键变化
◦单卡batch_size：代码中batch_size=32，指每个GPU每批处理32个样本
◦全局batch_size：单卡batch_size × world_size  = 32 × 4 = 128
◦每个epoch的总batch数：总样本数 / 全局batch_size = 10000 / 128 ≈ 78
◦每个rank的batch数：每个rank的样本数 / 单卡batch_size = 2500 / 32 ≈ 78（ 和总batch数一致）

1.
  错误：分布式训练时 batch_size是全局的
  正确：PyTorch DDP中batch_size永远是单卡的，全局大小要自己乘以 GPU数
2.
  错误：忘记调用sampler.set_epoch(epoch)

LoRA Rank 说明
◦定义：LoRA通过在注意力层旁插入两个小矩阵 A(r×d)和B(d×r)来训练，r就是这两个矩阵的秩
◦作用：控制LoRA的表达能力和参数量
•r越小：参数量越少，训练越快，显存占用越低，但表达能力越弱
•r越大：表达能力越强，但参数量和训练成本也越高
