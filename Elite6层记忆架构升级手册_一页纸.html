# WarmStore 核心骨架
import lancedb, os, time, json
from openai import OpenAI

class WarmStore:
    def __init__(self, db_path="~/.agent-memory/lancedb"):
        self.db = lancedb.connect(os.path.expanduser(db_path))
        self.embedder = OpenAI()
        try:
            self.table = self.db.create_table("memory", schema=[
                ("id","varchar"), ("text","text"), ("vector","vector,1536"),
                ("metadata","json"), ("created_at","float"), ("access_count","int"),
            ])
        except: self.table = self.db.open_table("memory")

    def add(self, text, metadata=None):
        vec = self._embed(text)
        self.table.add([{"id":f"m_{int(time.time())}","text":text,
            "vector":vec,"metadata":metadata or {},
            "created_at":time.time(),"access_count":0}])

    def search(self, query, limit=5):
        vec = self._embed(query)
        results = self.table.search(vec).limit(limit).to_list()
        for r in results: r["access_count"] += 1  # LRU
        return results

    def _embed(self, text):
        resp = self.embedder.embeddings.create(
            model="text-embedding-3-small", input=text[:8000])
        return resp.data[0].embedding
