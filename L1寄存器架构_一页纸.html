4 CLI 速查
elite_register.py set TASK active_task "构建L1寄存器"     # 写入
elite_register.py set TEMP cache_val "768" --ttl 3600    # 带TTL写入
elite_register.py get TASK active_task                     # 读取
elite_register.py list                                    # 列出所有
elite_register.py list --slot TASK                        # 列出指定槽位
elite_register.py expire                                  # 清除过期项
elite_register.py promote TASK active_task                # 晋升到L2
elite_register.py snapshot                                # 完整快照
elite_register.py status                                  # 状态概览
elite_register.py session new "新任务描述"                 # 开启新会话
elite_register.py clear --slot TEMP                      # 清空槽位
